Produce dataframe with coordinates of not cycled route of tour, ie between end of previous stage and start of current stage.

tour_not_cyc_stage <- tour_data_corr_gps %>%
  arrange(stage_id) %>%
  #select(year, start_lat, start_long, end_lat, end_long) %>%
  group_by(year) %>%
  mutate(end_prev_lat = if_else(stage_id == min(stage_id), 1E9, lag(end_lat)),
         end_prev_long = if_else(stage_id == min(stage_id), 1E9, lag(end_long))) %>%
  ungroup() %>%
  filter(end_prev_lat != 1E9) 

tour_not_cyc_stage <- tour_not_cyc_stage %>%
  select(year, end_prev_lat, end_prev_long, start_lat, start_long)

Add tour_progress variable to tour stages data for colour gradient in chart.

tour_data_corr_gps <- tour_data_corr_gps %>%
  group_by(year) %>%
  mutate(tour_progress = stage_number / max(stage_number)) %>%
  ungroup()

Define function plot_tour_ind to plot tour routes for years specified in year_start and year_end. Solid lines are the cycled stages. Dashed lines connect the end of one stage to start of another (ie not cycled).

plot_tour_ind <- function(year_start, year_end) {
tour_subset_loc <- tour_all_loc %>%
  filter(year >= year_start & year <= year_end)

tour_subset_stage <- tour_data_corr_gps %>%
  filter(year >= year_start & year <= year_end)

tour_subset_not_cyc <- tour_not_cyc_stage %>%
  filter(year >= year_start & year <= year_end)

# Calculate mean stage number for midpoint of colour gradient
mean_stage <- tour_subset_stage %>%
  summarise(mean_stage = mean(stage_number))

mean_stage <- mean_stage[[1, 1]]

# Plot route for this year(s)
tours_plot <- borders_plot +
  # Plot start and end locations of tour
  geom_point(data = tour_subset_loc,
             aes(x = long, y = lat),
             size = 0) +
  # Plot cycled parts of tour as solid lines
  geom_segment(
    data = tour_subset_stage,
    aes(
      x = start_long,
      y = start_lat,
      xend = end_long,
      yend = end_lat,
      colour = tour_progress
    ),
   # arrow = arrow(length = unit(0.05, "npc"), angle = 20),
    size = 1.5
   ) +
  # Plot not cycled route between end of previous and start of next stage as dashed line
  geom_segment(
    data = tour_subset_not_cyc,
    aes(
      x = end_prev_long,
      y = end_prev_lat,
      xend = start_long,
      yend = start_lat
    ),
    size = 0.5,
    colour = "#2E2D2D",
    linetype = 3
  ) +
  scale_colour_gradient(low = "green", high = "#01061d") +
  # scale_color_gradient2(low = "green2", mid = "black", high = "red2", midpoint = mean_stage) +
  facet_wrap( ~ year)

tours_plot
}
